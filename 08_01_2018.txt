//#define _CRT_SECURE_NO_WARNINGS
#include<windows.h>
#include<stdio.h>
#include<conio.h>
#include<stdlib.h>
#include<time.h>

int main()
{
	srand(time(NULL));
	const int N = 100, M = 200;
	int array[N][M], n, m, a = 1, b = 10, min, max;

	char q;

	do
	{
		system("cls");
		printf("\nVvedite kolichestvo strok v massive ot 1 do %d: ", N);
		scanf("%d", &n);

		if(n < 1 || n > N)
		{
			system("cls");// очистка консоли
			printf("\n\n\tERROR!!!\n");
			//_getch();
			fflush(stdin);
			scanf("%c", &q);
		}
	}
	while(n < 1 || n > N);

	do
	{
		printf("\nVvedite kolichestvo stolbcov v massive ot 1 do %d: ", M);
		scanf("%d", &m);

		if(m < 1 || m > M)
		{
			system("cls");
			printf("\n\n\tERROR!!!\n");
			_getch();
		}
	}
	while(m < 1 || m > M);
	
	for(int i = 0; i < n; i++)
	{
		for(int j = 0; j < m; j++)
		{
			array[i][j] = a + rand() % (b - a + 1);
		}
	}
	
	printf("\n\n--------------------------------\n\n");

	//вывод в виде таблицы
	for(int i = 0; i < n; i++)
	{
		for(int j = 0; j < m; j++)
		{
			printf("%-5d", array[i][j]);
		}
		printf("\n\n\n");
	}

	min = array[0][0];
	max = array[0][0];

	for(int i = 0; i < n; i++)
	{
		for(int j = 0; j < m; j++)
		{
			if(array[i][j] > max)
			{
				max = array[i][j];
			}

			if(array[i][j] < min)
			{
				min = array[i][j];
			}
		}
	}


	for(int i = 0; i < n; i++)
	{
		for(int j = 0; j < m; j++)
		{
			if(array[i][j] == max)
			{
				printf("max = array[%d][%d] = %d\n", i, j, max);
			}

			if(array[i][j] == min)
			{
				printf("min = array[%d][%d] = %d\n", i, j, min);
			}
		}
	}
	
	printf("\n\n\n\n\n\n\t\t\tThe end");
	_getch();
	return 0;
}
