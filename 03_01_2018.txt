#include<windows.h>
#include<stdio.h>
#include<conio.h>
#include<math.h>

int main()
{
	const int N = 100;
	int array[N], n, max, min, imax, imin;

	do
	{
		printf("\nVvedite razmer massiva ot 1 do %d: ", N);
		scanf("%d", &n);

		if(n < 1 || n > N)
		{
			system("cls");
			printf("\n\n\tERROR!!!\n");
			_getch();
		}
	}
	while(n < 1 || n > N);

	//заполняем массив данными
	for(int i = 0; i < n; i++)
	{
		printf("\nVvedite array[%d] = ", i);
		scanf("%d", &array[i]);
	}

	//вывод массива в строку
	printf("\n\n");
	for(int i = 0; i < n; i++)
	{
		printf("%d ", array[i]);
	}
	printf("\n\n");

	max = array[0];
	imax = 0;
	min = array[0];
	imin = 0;

	for(int i = 1; i < n; i++)
	{
		if(array[i] > max)
		{
			max = array[i];
			imax = i;
		}

		if(array[i] < min)
		{
			min = array[i];
			imin = i;
		}
	}

	printf("\nmax = array[%d] = %d\n", imax, array[imax]);
	printf("\nmin = array[%d] = %d\n", imin, array[imin]);
		
	printf("\n\n\n\n\n\n\t\t\tThe end");
	_getch();
	return 0;
}
